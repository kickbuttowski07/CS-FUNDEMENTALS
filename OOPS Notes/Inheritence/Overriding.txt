#include <iostream>

class Base
{
public:
    Base() { }

    void identify() const { std::cout << "Base::identify()\n"; }
};

class Derived: public Base
{
public:
    Derived() { }

    void identify() const { 
        std::cout << "Derived::identify()\n"; 
        Base :: identify(); // because of scope resolution, it calls Base identify member function.
        identify(); // this will call self / local member function, leads to infinite loop.
    }
};

int main()
{
    Base base {};
    base.identify();

    Derived derived {};
    derived.identify();

    return 0;
}
